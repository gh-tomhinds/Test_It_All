def fInit_Clients():
# - - - - - - - - - - - - - - - - - - - - - - - - - #

    # start with manually entered client, then load other clients from file
    clientList = ["Agawam"]
    allClis = csv.DictReader(open(Settings.cliFile))

    for oneCli in allClis:
        clientList.append(oneCli["NN1"])
    clientList.sort()

    return(clientList)
